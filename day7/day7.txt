CALLBACK FUNCTION - giving a function as an argument inside another function

forEach() - works as callback function (IIFE)
no relation with indexing
sbke liye baari baari se chalega...
Example- 
    let names=["Sonu","Monu","Birju","Kaju","Raju"]
    console.log(names)
    names.forEach(function(name){
        console.log(name);
    });

// for using 2 values - first value as the element and second value as the index
names.forEach((element,index)=>{
    console.log(index,"---",element);
});

for using for of - - - - -
for(let name of names){
    console.log(name);
}

SO, THERE ARE TOTAL 4 METHODS USING WHICH TRAVERSING IN ARRAYS CAN BE DONE
1-      // forEach() using callback function
        names.forEach(function(name){
            console.log(name);
        });

2-      // forEach() using arrow function
        names.forEach((n)=>{
            console.log(n);
        });

3-      // for using for of
        for(let name of names)
            console.log(name);

4-      // for using 2 values
        names.forEach((element,index)=>{
            console.log(index,"---",element);
        });

SLICE(start, end-1) FUNCTION -  original array is not changed
// abc = (11) [11, 22, 22, 5, 33, 4, 5, empty × 3, 'Bholu']

//DEALING WTH NEGATIVE INDEX USING SLICE
        let separate=abc.slice(5,-1)
        console.log(separate)

//DELETE USING SPLICE
        let splAgain=xyz.splice(3,2)
        console.log("Spliced values: "+splAgain);        //99,111
        console.log("New data after removal: "+xyz)      //11,22,88

//INSERT USING SPLICE
        let insert=xyz.splice(1,0,98,87,76,64);
        console.log("Removed value: "+insert);           //empty
        console.log("New data after insertion: "+xyz);   //11,98,87,76,64,22,88

//UPDATE USING SPLICE
        let update=xyz.splice(1,3,"Bholu","Birju","Sholu");
        console.log("Updated value: "+update);           //98,87,76
        console.log("New data after updation: "+xyz);    //11,Bholu,Birju,Sholu,64,22,88

returns the values in another array
    let list=[11,22,33,44,55,66]
    // Map Function of array - works as forEach but is faster
    let double= list.map((n)=>{
        // return n*2;               //returns [22,44,66,88,110,132]
        if(n%2==0)
            return n*2               // return [undefined,44,undefined,88,undefined,132]

let l=[11,22,33,44,55,66];
// FILTER FUNCTION - works on the basis of a condition
does not return true or false
instead, return another array based on what condition is true
let even=l.filter((num)=>num%2==0);

//FIND FUNCTION - returns in a variable, the first element that satisfies a given condition 
returns undefined if no such element
    let finding=l.find((num)=>num%2==0);  
    console.log(finding);

//FINDINDEX Function - returns in a variable, the index of first element that satisfies a given condition 
returns -1 if no such index
    let findIndex = l.findIndex((num)=>num>40);
    console.log(findIndex);
