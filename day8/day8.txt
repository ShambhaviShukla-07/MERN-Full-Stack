REDUCE(function(),initialValue) - (if no initial value is given then it takes the first value of the array as the initial value)
Gives the summary of any array....
can do sum of the elements of array
can give the average of the elements of array
can give sum based on some conditions(like sum of numbers at even position)
        let a =[9,2,3,4,5];
        let result=a.reduce(function(sum,num){   //sum-initial value(0) and num is every element inside array a
            return sum + num;
        },0);      //Initial values starts from 0
        console.log(result);           // 23
        //Accumulator has an initial value of 0 and then starts storing everything inside it
        let res=a.reduce((s,num)=>{
            return s+num
        },0);
        console.log(res);              // 23

        let r=a.reduce(function(sum,no){
            if(no%2==0)
                return sum+no
            return sum 
        });                            //INITIAL value - 9 (the first value in the array)
        console.log(r);                // 15

SPREAD OPERATOR - jb array ko todna hota hai
Data ko update krne k liye bhi use hota hai SPREAD OPERATOR

jb kisi array ya object ko kisi aur variable me assign krte hain to array ka data nhi jaata usme...balki array ka reference jaata hai
to ek ko change kiye to doosra bhi change jo jaayega
    let a=[11,22];
    let b=a;
    console.log(a);   //[11,22]
    console.log(b);   //[11,22]

    b[0]="Aditya";
    console.log(b);   //[11,22,"Aditya"]
    console.log(a)    //[11,22,"Aditya"]

SPREAD OPERATOR array k reference ko store krne k bajaaye copy store krne k liye bhi use hota hai taaki ek array change ho to uske parallelly uska original array change na ho
    let c=[33,44];
    let d= [...c];    //Spread Operator  
    c[0]="Anuj";
    console.log(c);               // ["Anuj",44]
    console.log(d);               // [33,44]
